function sum(...args1) {
  // 求和
  let x = args1.reduce((sum, next) =>sum+next)
  return function(...args2) {
    if (args2.length == 0) return x;
    // 求和
    let y = args2.reduce((sum, next) =>sum+next)
    return sum(x+y)
  }
}

function debounce(fun,time) {
    let timeout; // 存放定时器
    return function () {
        clearTimeout(timeout);
        timeout = setTimeout(() => {
            fun()
        }, time);
    };
}
// 触发立即执行一次
function debounce(func,wait) {
    let timeout;
    return function () {
        if (timeout) clearTimeout(timeout);
        let callNow = !timeout;
        timeout = setTimeout(() => {
            timeout = null;
        }, wait)
        if (callNow) func()
    }
}

function deepClone (obj){
    if (obj===undefined || obj===null) return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof RegExp) return new RegExp(obj);
    if (typeof obj !== 'object') return obj;
    let cloneObj = new obj.constructor;
    for (const key in obj) {
        cloneObj[key] = deepClone(obj[key]);
    }
    return cloneObj;
}

// 快排 小到大
function quickSort(arr,left,right) {
if(left>right) return
let l=left,r=right,baseValue=arr[l]
while (l<r) {
    while (l<r && arr[r]>=baseValue) {
        r--
    }
    while (l<r && arr[l]<=baseValue) {
        l++
    }
    if(l<r){
        let temp = arr[r]
        arr[r] = arr[l]
        arr[l] = temp
    }
}
arr[left] = arr[l]
arr[l] = baseValue
quickSort(arr,l+1,right)
quickSort(arr,left,l-1)
return arr
}

#### 观察者模式

```
class EventEmitter {
  constructor() {
    this.cache = {}
  }
//开启监听,事件名,事件触发的函数
  on(name, fn) {
    if (this.cache[name]) {
      this.cache[name].push(fn)
    } else {
      this.cache[name] = [fn]
    }
  }
//取消监听，取消订阅的事件名、对应的函数
  off(name, fn) {
    const tasks = this.cache[name]
    if (tasks) {
      const index = tasks.findIndex((f) => f === fn || f.callback === fn)
      if (index >= 0) {
        tasks.splice(index, 1)
      }
    }
  }
//事件发布
  emit(name) {
    if (this.cache[name]) {
      // 创建副本，如果回调函数内继续注册相同事件，会造成死循环
      const tasks = this.cache[name].slice()
      for (let fn of tasks) {
        fn();
      }
    }
  }

  emit(name, once = false) {
    if (this.cache[name]) {
      // 创建副本，如果回调函数内继续注册相同事件，会造成死循环
      const tasks = this.cache[name].slice()
      for (let fn of tasks) {
        fn();
      }
      if (once) {
        delete this.cache[name]
      }
    }
  }
}

#### promise

```
class MyPromise {
  constructor(func) {
    this.status = 'pending'
    this.value = null
    this.resolvedTasks = []
    this.rejectedTasks = []
    this._resolve = this._resolve.bind(this)
    this._reject = this._reject.bind(this)
    try {
      func(this._resolve, this._reject)
    } catch (error) {
      this._reject(error)
    }
  }

  _resolve(value) {
    setTimeout(() => {
      this.status = 'fulfilled'
      this.value = value
      this.resolvedTasks.forEach(t => t(value))
    })
  }

  _reject(reason) {
    setTimeout(() => {
      this.status = 'reject'
      this.value = reason
      this.rejectedTasks.forEach(t => t(reason))
    })
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      this.resolvedTasks.push((value) => {
        try {
          const res = onFulfilled(value)
          if (res instanceof MyPromise) {
            res.then(resolve, reject)
          } else {
            resolve(res)
          }
        } catch (error) {
          reject(error)
        }
      })
      this.rejectedTasks.push((value) => {
        try {
          const res = onRejected(value)
          if (res instanceof MyPromise) {
            res.then(resolve, reject)
          } else {
            reject(res)
          }
        } catch (error) {
          reject(error)
        }
      })
    })
  }

  catch(onRejected) {
    return this.then(null, onRejected);
  }
}
